创建store
```js
import { createSlice } from "@reduxjs/toolkit"

export const Reducer = createSlice({
    name: "test",
    initialState: {},
    reducers: {
        set: (state, action) => {
            state = action?.payload || {}
        },
	    clear: (state, action) => {
            state = {}
        },
    },
})

export const { set, clear } = Reducer.actions

export default Reducer.reducer
```
