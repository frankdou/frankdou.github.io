// 推荐以下明确的class名，以便Tailwind Scanner可以检索，从而打包进样式包
const bgColor = isRed ? 'bg-red-500' : 'bg-blue-500';

// tailwind.config.js safelist明确添加需要打包进的class
module.exports = {
  // ... other configs
  safelist: [
    'bg-red-500',
    'bg-blue-500',
    'bg-green-500',
    {
      // You can also use regular expressions for patterns
      pattern: /p-(2|4|6)/, // Safelists p-2, p-4, and p-6
      variants: ['sm', 'lg'], // Also safelists sm:p-2, sm:p-4, etc.
    },
  ],
  // ...
}
