不要使用内置的`Readonly<T>`，自己实现一个。

```ts
type MyReadonly<T> = {
    readonly[K in keyof T]: T[K]
}

import type { Equal, Expect } from '@type-challenges/utils'

type cases = [
  Expect<Equal<MyReadonly<Todo1>, Readonly<Todo1>>>,
]

interface Todo1 {
  title: string
  description: string
  completed: boolean
  meta: {
    author: string
  }
}
```
