const sortByMax = (arr) => {
  if (!arr.length) return [];

  const counters = arr.reduce((results, item) => {
    if (results[item]) {
      results[item]++;
    } else {
      results[item] = 1;
    }
  
    return results
  }, {})
  
  return Object.keys(counters).sort(((a,b ) => counters[b] - counters[a]))
}

console.log(sortByMax(['a', 'ab', 'abc', 'abcd', 'ab', 'abc', 'abcd', 'abc', 'abcd', 'abcd']));
