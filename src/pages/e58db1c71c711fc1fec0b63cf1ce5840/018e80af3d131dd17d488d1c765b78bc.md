---
title: canvas获取文字所占尺寸
size: 636
---
```js
const canvas = document.getElementById('myCanvas');
const ctx = canvas.getContext('2d');

// 设置字体样式，这一步非常重要！
ctx.font = '48px serif';

const text = '你好，世界！';

// 使用 measureText() 获取文字尺寸
const metrics = ctx.measureText(text);

// 获取宽度
const textWidth = metrics.width;

// 获取实际高度（从 ascent 到 descent）
// emHeightAscent
const textHeight = metrics.actualBoundingBoxAscent + metrics.actualBoundingBoxDescent;

console.log(`文字: "${text}"`);
console.log(`宽度: ${textWidth.toFixed(2)} px`);
console.log(`高度: ${textHeight.toFixed(2)} px`);
```
