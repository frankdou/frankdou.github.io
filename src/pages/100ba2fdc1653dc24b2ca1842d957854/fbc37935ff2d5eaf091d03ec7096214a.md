##### 寄生组合式继承

  - 继承属性：new子类的时候，通过call调用主类的构造函数

  ```js
  function Sub(name,age){
      //属性继承
      Base.call(this, name)
      this.age = age
  }
  ```

  - 继承原型方法

  ```javascript
  function inheritPrototype(child, parent){
    // 创建父类原型的一个副本
    let prototype = Object.create(parent.prototype)
  
    //如果希望继承多个对象
    //Object.assign(MyClass.prototype, OtherSuperClass.prototype);
    	
    // constructor属性设置
    prototype.constructor = child
  
    // 赋值给子类的原型
    child.prototype = prototype
  }
  ```

- 完整示例

```js
function inheritPrototype(child, parent){
  // 创建父类原型的一个副本
  let prototype = Object.create(parent.prototype)

  // constructor属性设置
  prototype.constructor = child

  // 赋值给子类的原型
  child.prototype = prototype
}

// 基类
function Animal(skills){
  this.skills = skills
}

Animal.prototype.getSkills = function(){
  console.log(this.skills)
}

// 子类
function Dog(skills, color){
  Animal.call(this, skills)
  this.color = color
}

// 继承原型
inheritPrototype(Dog, Animal) 

// 添加子类方法
Dog.prototype.getColor = function(){
  console.log(this.color)
}

const dog = new Dog(['run', 'eat'], 'black')

dog.getSkills()
dog.getColor()
```

##### ES6（extends）

```js
class Animal { 
  constructor(name) {
    this.name = name;
  }
  speak() {
    console.log(`${this.name} makes a noise.`);
  }
}

class Dog extends Animal {
  constructor(name) {
    super(name); // 调用父类构造函数并传入name参数
  }

  speak() {
		super.speak(); // 调用父类的方法
    console.log(`${this.name} barks.`);
  }
}

const instance = new Dog('Mitzie');
instance.speak();
//Mitzie makes a noise.  Mitzie barks.
```
