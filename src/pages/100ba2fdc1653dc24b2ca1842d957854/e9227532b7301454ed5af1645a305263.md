##### for
js是单线程语言，每次循环都是在主线程中执行的，循环体中的代码会在前一个元素处理完成之后才会执行下一个元素的处理，按照定义的顺序依次执行

function getPathById(tree, id) {
	const result = []
	function findPath(node) {
		console.log(node, '=-=-');
		if (node.id === id) {
			result.push(node.id)
			return true
		}
		
		if (node.children && node.children.length) {
			for (let i = 0; i < node.children.length; i++) {
				const target = node.children[i]

				if (findPath(target)) {
					result.push(node.id)
					return true
				}
			}
		}
		return false
	}

	for (let i = 0; i < tree.length; i++) {
		if (findPath(tree[i])) {
			break
		}
	}

	return result.reverse()
}

console.log(getPathById(sample, 333))

循环适合用于需要手动控制循环的情况，例如需要使用条件语句来控制循环
###### 中止循环
```js
const arr = [1, 2, 3, 4, 5];

for (let i = 0; i < arr.length; i++) {
  if (arr[i] === 3) {
    break;
  }
  console.log(arr[i]);
}
```
###### 跳过当前元素
```js
const arr = [1, 2, 3, 4, 5];

for (let i = 0; i < arr.length; i++) {
  if (arr[i] === 3) {
    continue;
  }
  console.log(arr[i]);
}
```

##### forEach
设计初衷是为了在数组的每个元素上执行相同的操作，它会在遍历整个数组之后自动停止，没有提供中断循环的API
可以通过使用 `some`，`find`等方式来做中断循环的操作
