```js
const fib = (number) =>  {
    let memoize = (number) => {
        let cache = {
            0: 0,
            1: 1
        };
        
        for (let i = 2; i <= number; i++) {
            // 下一个key的value，为上两个值的和
            cache[i] = cache[i - 1] + cache[i - 2];
        }

        return cache[number];
    }
	
    if (number <= 1) {
        return number;
    }

    return memoize(number);
}

console.log(fib(8));
```
