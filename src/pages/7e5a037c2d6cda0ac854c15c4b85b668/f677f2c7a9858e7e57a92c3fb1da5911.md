const bubbleSort = (arr) => {
    const length = arr.length;

    for(let i = 0; i < length - 1; i++) {
				// 多 -1 是因为后面的if判断会存在过载的状态
        for (let n = 0; n < length - i - 1; n++) {
            // 如果大于下一位，则切换位置
            if (arr[n] > arr[n + 1]) {
                [arr[n], arr[n + 1]] = [arr[n + 1], arr[n]]
            }
        }
    }

    return arr;
}
