路由处理逻辑复用
```js
// handler方式
const sharedHandler = (req, res) => {
  const resource = req.path.split('/')[2]
  res.json({ resource, data: getData(resource) });
};

app.get('/api/aaa', sharedHandler);
app.get('/api/bbb', sharedHandler);

// 数组路由
app.get(['/api/aaa', '/api/bbb'], (req, res) => {
  const resource = req.path.split('/')[2];

  res.json({ resource, data: getData(resource) });
});
```
