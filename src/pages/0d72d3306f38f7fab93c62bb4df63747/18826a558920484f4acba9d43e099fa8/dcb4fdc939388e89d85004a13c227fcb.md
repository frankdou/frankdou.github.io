date: 2022-05-12T12:27:37.827Z
size: 377
---
```javascript
computed: {
    project: {
        get() {
            return this.$attrs.project;
        },
        set(value) {
            this.$emit('update:project', value);
        }
    },
    namespace: {
        get() {
            return this.$attrs.namespace;
        },
        set(value) {
            this.$emit('update:namespace', value);
        }
    },
}
```
