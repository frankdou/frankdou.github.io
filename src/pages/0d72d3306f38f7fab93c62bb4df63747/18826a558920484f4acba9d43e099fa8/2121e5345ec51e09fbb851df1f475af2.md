```vue
<template>
    <div />
</template>

<script>
import $ from 'jquery';

export default {
    data() {
        return {
            dIframe: null,
        };
    },
    computed: {
        sUrl() {
            return 'http://xxxxxx';
        },
    },
    mounted() {
        this.init();
    },
    methods: {
        init() {
            this.dIframe = $(`<iframe src=${this.sUrl} class='iframe' />`);

            // iframe onload event
            this.dIframe.on('load', this.onLoaded);

            // append iframe
            $(this.$el).append(this.dIframe);
        },
        onLoaded() {
            // send data to iframe
            this.dIframe[0].contentWindow.postMessage('loaded', this.sUrl);
        },
    },
};
</script>
